#include<stdio.h>
int main(){
int n;
printf("Enter the no of elements: ");
scanf("%d",&n);
int a[n],temp;
for(int i=0;i<n;i++){
scanf("%d",&a[i]);
}
for(int i=1;i<n;i++){
temp=a[i];
j=i-1;
while(j<=0 and a[j]<temp){
   a[j+1]=a[j];
   j--;
}
a[j+1]=temp;
printf("After Sorting: ");
for(i=0;i<n;i++){
printf("%d",a[i]);
}

}
